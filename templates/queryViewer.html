select PROJECT.NAME [Submission]
, Format(DATE_CREATED, 'dd MMM yyyy') [Date Created]
, LIST_ENTRY.VALUE [Submission Reason]
, Case when OWNER = 'AHHC' then 'Health Certification' else 'Diagnostic' END [Case Manager]
, CONTRACT_QUOTE.DESCRIPTION [Contract]
, CASE WHEN MONTH(DATE_CREATED) > 6 THEN LTRIM(STR(YEAR(DATE_CREATED))) + '/' + LTRIM(STR(YEAR(DATE_CREATED) + 1))
ELSE LTRIM(STR(YEAR(DATE_CREATED) - 1)) + '/' + LTRIM(STR(YEAR(DATE_CREATED))) END [Fiscal Year]
, TEXT_ID [Text ID]
, TEST.TEST_NUMBER [Test Number]
, RESULT.ANALYSIS
, RESULT.REPORTED_NAME
, NUMERIC_ENTRY

from PROJECT join SAMPLE on PROJECT.NAME = SAMPLE.PROJECT
join TEST on SAMPLE.SAMPLE_NUMBER = TEST.SAMPLE_NUMBER
join RESULT on TEST.TEST_NUMBER = RESULT.TEST_NUMBER
join CONTRACT_QUOTE on T_CONTRACT_NUMBER = CONTRACT_QUOTE_NO
join LIST_ENTRY on C_SUBMISSION_REASON = LIST_ENTRY.NAME and LIST_ENTRY.LIST = 'AH_SUB_REASON'

where RESULT.REPORTED_NAME = 'Number of Necropies'
and RESULT.STATUS = 'A'
and TEST.STATUS <> 'X' and SAMPLE.STATUS <> 'X'

UNION ALL

select PROJECT.NAME [Submission] , Format(DATE_CREATED, 'dd MMM yyyy') [Date Created] , IsNULL(LIST_ENTRY.VALUE, REASON_1) [Submission Reason] , Case when OWNER = 'AHHC' then 'Health Certification' else 'Diagnostic' END [Case Manager] , CONTRACT_QUOTE.DESCRIPTION [Contract] , CASE WHEN MONTH(DATE_CREATED) > 6 THEN LTRIM(STR(YEAR(DATE_CREATED))) + '/' + LTRIM(STR(YEAR(DATE_CREATED) + 1)) ELSE LTRIM(STR(YEAR(DATE_CREATED) - 1)) + '/' + LTRIM(STR(YEAR(DATE_CREATED))) END [Fiscal Year] , TEXT_ID [Text ID] , TEST.TEST_NUMBER [Test Number] , 'AH_' + TEST.ANALYSIS [Analysis] , RESULT.REPORTED_NAME , NUMERIC_ENTRY from VICGOV_LW_VET.dbo.PROJECT join VICGOV_LW_VET.dbo.SAMPLE on PROJECT.NAME = SAMPLE.PROJECT join VICGOV_LW_VET.dbo.TEST on SAMPLE.SAMPLE_NUMBER = TEST.SAMPLE_NUMBER join VICGOV_LW_VET.dbo.RESULT on TEST.TEST_NUMBER = RESULT.TEST_NUMBER join VICGOV_LW_VET.dbo.CONTRACT_QUOTE on PROJECT.CONTRACT_NUMBER = CONTRACT_QUOTE_NO left join VICGOV_LW_VET.dbo.LIST_ENTRY on REASON_1 = LIST_ENTRY.NAME and LIST_ENTRY.LIST = 'REASON_1' where RESULT.REPORTED_NAME = 'Number of Necropies' and RESULT.STATUS = 'A' and TEST.STATUS <> 'X' and SAMPLE.STATUS <> 'X' and DATE_CREATED >= '2014-07-01' ORDER BY [Fiscal Year], Submission, [Text ID]
